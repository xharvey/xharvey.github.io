    def convert(filename): 
	try: 
		content=codecs.open(filename,'r').read()

遍历目录所有文件

    def listFiles(dirPath):
	fileList=[]
	for root,dirs,files in os.walk(dirPath):
		for fileObj in files:
			fileList.append(os.path.join(root,fileObj))
	return fileList
